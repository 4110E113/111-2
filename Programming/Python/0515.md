def cheak_grade(score):
    if score >=90:
        print('A')
    elif score >=80:
        print('B')
    elif score >=70:
        print('C')
    elif score >=60:
        print('D')
    elif score>=50 and score<60:
        print('E')
    else:
        print('F')

while True:
    try:
        score=int(input("score: "))
        break
    except ValueError:
        print("Value Error")
    
cheak_grade(score)

```py
my_string = "kun shan"

num_uppercase = 0

for char in my_string:

    if char.isupper():
    
        num_uppercase += 1
       
       
print("original string: " + my_string)

print("number of uppercase letters: " + str(num_uppercase))

revesed_string = ""

for i in range(len(my_string)-1,-1,-1):
    revesed_string += my_string[i]
    
print("reversed string: " + revesed_string)
```
